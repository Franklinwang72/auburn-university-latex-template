% ======================================================================
%  ACCESSIBILITY METADATA -- MUST BE THE VERY FIRST LINE OF THE FILE
%  (before \documentclass, before everything except these comments).
%  This is what turns on tagged-PDF output for WCAG 2.1 AA / PDF/UA-1.
%  Do NOT delete it, and keep compiling with LuaLaTeX.
% ======================================================================
\DocumentMetadata{
  lang        = en-US,  % document language (screen readers announce it)
  pdfstandard = ua-1,   % PDF/UA-1 accessibility conformance target
  pdfversion  = 1.7,    % FORCE the file header to %PDF-1.7 so veraPDF's
                        % PDF/UA-1 header rule (ISO 14289-1 clause 6.1)
                        % passes, EVEN THOUGH the tagging engine still
                        % embeds MathML via PDF-2.0 associated files (/AF).
                        % This is a deliberate mismatch: header says 1.7,
                        % body carries 2.0 MathML. It makes veraPDF's UA-1
                        % profile pass while keeping the MathML layer.
                        % NOTE: this is non-standard (a 1.7 file is not
                        % supposed to use /AF); a stricter validator may
                        % flag it. Remove this line for a clean file.
  tagging     = on,     % build the tag structure tree + MathML
}

%%      -> the latest. The project is fully self-contained: every font

%%      Atkinson Hyperlegible Mono for code) is bundled in fonts/
%%      together with its OFL licence.
